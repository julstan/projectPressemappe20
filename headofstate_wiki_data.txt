  ?person wdt:P4293 ?pm20id.
  OPTIONAL {?person wdt:P569 ?birthday} .
  OPTIONAL {?person wdt:P570 ?deathday }.
  OPTIONAL {?person wdt:P18 ?bild } .
  ?person wdt:P39 ?posi.
  ?posi wdt:P279 wd:Q2285706.
  OPTIONAL { ?person wdt:P580 ?starttime. }
  OPTIONAL { ?person wdt:P582 ?endtime. }
  SERVICE wikibase:label { bd:serviceParam wikibase:language "de,en". }

SELECT ?pm20idLabel ?person ?personLabel ?birthday ?deathday ?picture ?position_heldLabel ?start ?end WHERE {
  ?person wdt:P4293 ?pm20id.
  OPTIONAL {?person wdt:P569 ?birthday} .
  OPTIONAL {?person wdt:P570 ?deathday }.
  OPTIONAL {?person wdt:P18 ?picture } .
  
  ?person p:P39 ?s.  # ?s is the the statement node
           ?s ps:P39  ?position_held.  # that simulates the predicate 
           ?position_held wdt:P279 wd:Q2285706.
           ?s pq:P580 ?start.  # this is an attribute (a qualifier)
           ?s pq:P582 ?end.
    
  SERVICE wikibase:label { bd:serviceParam wikibase:language "de,en". }
 } ORDER BY ?pm20idLabel
